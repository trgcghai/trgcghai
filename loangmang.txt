int dx[8] {-1, -1, -1, 0, 0, 1, 1, 1};
int dy[8] {-1, 0, 1, -1, 1, -1, 0, 1};
int n,m, a[100][100];
int dem = 0;

void loang(int i, int j, int x) {
    dem++;
    a[i][j] = 1000;
    for (int k = 0; k < 8; k++) {
        int i1 = i + dx[k];
        int j1 = j + dy[k];
        if (i1 > 0 && i1 <= n && j1 > 0 && j1 <= m && a[i1][j1] == x) {
            loang(i1, j1, x);
        } 
    }
}
